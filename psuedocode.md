IsMember = str(input("Are you currently a member?"))
Hours = int(input("How many hours would you like to book?"))

StandardChargeRate = 10
MemberChargeRate = 7

function CalculatePrice():
    if IsMember == "Yes" and :
        return Hours X MemberChargeRate
    else
        return Hours X StandardChargeRate

print(f'Your total cost is: {CalculatePrice()}')
